{
              h= h%12
              let hour =(h*360)/12 +(m*360)/(12*60)
              let min = (m*360)/60
              let angle = Math.abs(hour-min);

              if(angle>180)

              }

var Reverse_Number = (N) => 
{
  let rev=0;
  while(N!==0){
    let digit=N%10;
    rev=rev*10+digit;
    N=Math.floor(N/10);
  }
  return rev;
};
